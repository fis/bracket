38  /** Constructs a new ring buffer. \p initial_size must be a power of 2. */

41  CHECK(!(initial_size & (initial_size - 1)));
42  }
43 
44  /** Destroys this buffer, freeing the memory. */
45  ~RingBuffer() {
46  delete[] data_;
47  }

49  /**
50  * Allocates space for \p push_size bytes in the queue.
51  *
52  * The caller is responsible for writing some data. Of the two BufferView objects that are
53  * returned, the first one is always valid, and has a size between 1 and \p push_size. The second
54  * BufferView is only valid if the first one is smaller than the requested size, in which case
55  * *its* size will be `push_size - first.size`. This happens when the requested range crosses the
56  * wrap-around point of the buffer.
57  */
58  std::pair<BufferView, BufferView> Push(BufferSize push_size);
59 
60  /** Allocates and writes a single byte. */
61  void PushChar(char c) {
62  if (used_ == size_) {
63  *Push(1).first.data = c;
64  } else {
65  BufferSize end = (first_byte_ + used_) & (size_ - 1);
66  data_[end] = c;
67  ++used_;
68  }
69  }
70 
71  /**
72  * Deallocates storage from the end of the buffer.
73  *
74  * This method can be useful if you need to Push() something for which you know an upper bound,
75  * but not the exact size. Call Push() with the upper bound size, write your contents, and then
76  * Unpush() the remaining space.
77  *
78  * Be aware that the Push() call will ensure there is sufficient free space for the entire length,
79  * potentially allocating memory. Only use this approach if the upper bound is relatively tight.
80  */
81  void Unpush(BufferSize size) {
82  CHECK(size <= used_);
83  used_ -= size;
84  if (!used_)
85  first_byte_ = 0;
86  }
87  /**
88  * Returns a view to the first \p size bytes of the queue.
89  *
90  * The argument must be at most #size(). As with Push(), the first returned view is always valid,
91  * and the second one only if the boundary is crossed. `first.size + second.size` will be equal to
92  * the requested size.
93  */
94  std::pair<BufferView, BufferView> Front(BufferSize size) {
95  CHECK(size <= used_);
96  return ViewFrom(first_byte_, size);
97  }

114  /** Constructs a view for a region. */
115  std::pair<BufferView, BufferView> ViewFrom(BufferSize start, BufferSize view_size) {
116  if (start + view_size <= size_) {
117  // no wrapping
118  return std::make_pair(BufferView(data_ + start, view_size), BufferView());
119  } else {
120  // two pieces
121  BufferSize first_piece = size_ - start;
122  return std::make_pair(
123  BufferView(data_ + start, first_piece),
124  BufferView(data_, view_size - first_piece));
125  }
126  }
