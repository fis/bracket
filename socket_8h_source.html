29  /**
30  * Initiates an asynchronous read operation.
31  *
32  * The Watcher::CanRead() method will be called when you should call Read() on the socket. A
33  * successful read (one that returns any bytes, even if less than requested) will finish the
34  * operation, and you must call StartRead() again if you want to keep reading.
35  */
36  virtual void StartRead() = 0;
37  /**
38  * Initiates an asynchronous write operation.
39  *
40  * The Watcher::CanWrite() method will be called when you should call Write() on the socket. A
41  * successful write (one that consumes any bytes, even if less than requested) will finish the
42  * operation, and you must call StartWrite() again if you still have more bytes to write.
43  */
44  virtual void StartWrite() = 0;
45 
46  /**
47  * Attempts to read from the socket, returning the number of bytes read.
48  *
49  * If there was no data immediately available, returns zero. This does not finish the read, so
50  * there is no need to call StartRead() again. The Watcher::CanRead() method will be called when
51  * you should try again.
52  *
53  * For a TLS socket, it's not safe to call this method except as a response to the
54  * Watcher::CanRead() callback. A TLS read may be left in a pending state (if renegotiation is
55  * triggered). The implementation will make sure no Watcher::CanWrite() callbacks are delivered
56  * until the pending read has finished.
57  */
58  virtual std::size_t Read(void* buf, std::size_t count) = 0;
59 
60  /**
61  * Attempts to write to the socket, returning the number of bytes written.
62  *
63  * If no bytes could be written without blocking, returns zero. This does not finish the write, so
64  * there is no need to call StartWrite() again. The Watcher::CanWrite() method will be called when
65  * you should try again.
66  *
67  * For a TLS socket, it's not safe to call this method except as a response to the
68  * Watcher::CanWrite() callback. A TLS write may be left in a pending state (if renegotiation is
69  * triggered). The implementation will make sure no Watcher::CanWrite() callbacks are delivered
70  * until the pending read has finished.
71  *
72  * Also for a TLS socket, if you have already tried to write some bytes, you can no longer change
73  * your mind afterwards. If the return value indicates that not all bytes were written, the next
74  * time you call this method, you must pass in the same contents, to avoid unpredictable
75  * behavior. Some of the bytes may already have been copied to the library data structures.
76  */
77  virtual std::size_t Write(void* buf, std::size_t count) = 0;

91  /**
92  * Called to indicate the connection is open.

94  * Either this method or ConnectionFailed() is guaranteed to be called once.
95  */
96  virtual void ConnectionOpen() = 0;
97  /**
98  * Called to indicate the connection timed out.
99  *
100  * Either this method or ConnectionOpen() is guaranteed to be called once.
101  */
102  virtual void ConnectionFailed(const std::string& error) = 0;
103  /**
104  * Called to indicate that you must read from the socket.
105  *
106  * This method is guaranteed to be called at least once (as long as the remote side sends any
107  * data) after issuing a Socket::StartRead() call. You should attempt to read some data with
108  * Socket::Read(). If the read does not return any bytes yet, this method will be called again
109  * later, at which point you should try again. If any bytes were read, this method will not be
110  * called any more, until you call Socket::StartRead() again.
111  */
112  virtual void CanRead() = 0;
113  /**
114  * Called to indicate that you must write to the socket.
115  *
116  * This method is guaranteed to be called at least once after issuing a Socket::StartWrite()
117  * call. You should attempt to write some data with Socket::Write(). If the write does not
118  * succeed in writing any data (returns 0), this method will be called again later, at which
119  * point you should try again. If any bytes were written, this method will not be called any
120  * more, until you call Socket::StartWrite() again.
121  */
122  virtual void CanWrite() = 0;
