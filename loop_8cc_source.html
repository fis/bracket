39 void Loop::ReadFd(int fd, FdReader* callback, bool owned) {
40  auto&& fd_info = GetFd(fd);
41  if (callback) {
42  CHECK(fd_info->reader.empty());
43  fd_info->reader.Set(callback, owned);
44  if (!pollfds_.empty()) {
45  struct pollfd *pfd = &pollfds_[fd_info->pollfd_index];
46  pfd->events |= POLLIN;
47  if (pfd->fd < 0) pfd->fd = -pfd->fd;
48  }
49  } else {
50  fd_info->reader.Clear();
51  if (fd_info->reader.empty() && fd_info->writer.empty()) {
52  fds_.erase(fd);
53  pollfds_.clear();
54  } else if (!pollfds_.empty()) {
55  pollfds_[fd_info->pollfd_index].events &= ~POLLIN;

125 void Loop::Poll() {
126  CHECK(!fds_.empty()); // TODO: timer support (via timerfd?)
127 
128  if (!fds_.empty() && pollfds_.empty()) {
129  pollfds_.resize(fds_.size());
130 
131  std::size_t pollfd_index = 0;
132  for (auto&& fd_item : fds_) {
133  int fd = fd_item.first;
134  Fd* fd_info = &fd_item.second;
135 
136  int events = 0;
137  if (!fd_info->reader.empty())
138  events |= POLLIN;
139  if (!fd_info->writer.empty())
140  events |= POLLOUT;
141 
142  fd_info->pollfd_index = pollfd_index;
143 
144  struct pollfd* pfd = &pollfds_[pollfd_index];
145  pfd->fd = events ? fd : -fd;
146  pfd->events = events;
147 
148  ++pollfd_index;
149  }
150  }
151 
152  int changed = poll_(pollfds_.data(), pollfds_.size(), -1);
153  if (changed == -1 && errno != EINTR)
154  throw base::Exception("poll", errno);
155 
156  if (changed > 0) {
157  // Calling the callbacks may add or remove descriptors, which
158  // could invalidate iterators to fds_ or pollfds_. Collect a
159  // separate list of file descriptors on which callbacks need to be
160  // invoked, first.
161 
162  std::vector<std::pair<int, bool /* write? */>> events;
163 
164  // TODO POLLNVAL?
165  for (const struct pollfd& pfd : pollfds_) {
166  if (pfd.revents & (POLLIN|POLLERR|POLLHUP))
167  events.emplace_back(pfd.fd, false);
168  if (pfd.revents & (POLLOUT|POLLERR))
169  events.emplace_back(pfd.fd, true);
170  }
171 
172  for (const auto& event : events) {
173  int fd = event.first;
174 
175  const auto fd_entry = fds_.find(fd);
176  if (fd_entry == fds_.end())
177  continue; // no longer relevant
178 
179  if (event.second)
180  fd_entry->second.writer.Call(&FdWriter::CanWrite, fd);
181  else
182  fd_entry->second.reader.Call(&FdReader::CanRead, fd);
183  }
184  }
185 }
