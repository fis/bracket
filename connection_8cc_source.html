126 void Connection::CanRead() {
127  std::size_t got;
128 
129  try {
130  got = socket_->Read(read_buffer_.data() + read_buffer_used_, read_buffer_.size() - read_buffer_used_);
131  } catch (const event::Socket::Exception& e) {
132  ConnectionLost(e.what());
133  return;
134  }
135 
136  if (got == 0)
137  return;
138  socket_->StartRead();
139 
140  read_buffer_used_ += got;
141 
142  // parse complete messages and pass them to listeners
143 
144  char *start = read_buffer_.data();
145  std::size_t left = read_buffer_used_;
146 
147  while (left > 0) {
148  // read next complete message at 'start'
149 
150  std::size_t msg_len = 0;
151  while (msg_len < left && msg_len < kMaxMessageSize && start[msg_len] != 10 && start[msg_len] != 13)
152  ++msg_len;
153 
154  if (msg_len == kMaxMessageSize || start[msg_len] == 10 || start[msg_len] == 13) {
155  // found a delimiter, or reached maximum message size
156  if (msg_len > 0) {
157  if (read_message_.Parse(start, msg_len))
158  HandleMessage(read_message_);
159  else
160  LOG(ERROR) << "Invalid IRC message"; /// \todo dump bytes?
161  }
162  } else {
163  // hit end of buffer with an incomplete message
164  break;
165  }
166 
167  // consume delimiters that followed the message
168 
169  start += msg_len;
170  left -= msg_len;
171  while (left > 0 && (*start == 10 || *start == 13)) {
172  ++start;
173  --left;
174  }
175  }
176 
177  // save any incomplete data for the next round
178 
179  read_buffer_used_ = left;
180  if (left > 0)
181  std::memmove(read_buffer_.data(), start, left);
182 
183  read_message_.Clear();
184 }

210 void Connection::Send(const Message& message) {
211  bool was_empty = write_queue_.empty();
212 
213  constexpr std::size_t kMaxContentSize = kMaxMessageSize - 2;
214 
215  auto buffer = write_buffer_.Push(kMaxContentSize);
216  std::size_t message_size = message.Write(buffer.first.data, buffer.first.size); // first half
217  std::size_t write_size = std::min(message_size, kMaxContentSize);
218 
219  if (write_size > buffer.first.size) {
220  // didn't fit, so do trailing half
221  CHECK(buffer.second.valid() && write_size <= buffer.first.size + buffer.second.size);
222  char tmp_buffer[write_size];
223  message.Write(tmp_buffer, write_size);
224  std::memcpy(buffer.second.data, tmp_buffer + buffer.first.size, write_size - buffer.first.size);
225  }
226 
227  if (write_size < kMaxContentSize)
228  write_buffer_.Unpush(kMaxContentSize - write_size);
229 
230  write_buffer_.PushChar(13);
231  write_buffer_.PushChar(10);
232 
233  int cost = 1000;
234  auto extra_cost_it = extra_cost.find(message.command());
235  if (extra_cost_it != extra_cost.end())
236  cost += extra_cost_it->second;
237 
238  write_queue_.emplace_back(write_size + 2, cost);
239 
240  LOG(VERBOSE) << "Added " << write_size + 2 << " bytes to the write queue (cost " << cost << ')';
241 
242  // TODO: don't try until a connection has been established
243  if (was_empty) // otherwise we're trying already
244  socket_->StartWrite();
245 }

247 void Connection::CanWrite() {
248  if (write_queue_.empty())
249  return; // actually, nothing to write
250 
251  // update credit estimate
252 
253  if (write_credit_ < kMaxWriteCredit) {
254  auto now = loop_->now();
255  int delta =
256  std::max(std::min(std::chrono::duration_cast<std::chrono::milliseconds>(now - write_credit_time_),
257  std::chrono::milliseconds(kMaxWriteCredit)),
258  std::chrono::milliseconds(0)).count();
259  write_credit_ = std::min(write_credit_ + delta, kMaxWriteCredit);
260  write_credit_time_ = now;
261  }
262 
263  // see how much we can write
264 
265  std::size_t can_write = 0;
266  int credit_left = write_credit_;
267 
268  for (auto&& msg : write_queue_) {
269  int cost = 10 * msg.first + msg.second;
270  if (cost > credit_left)
271  break;
272  can_write += msg.first;
273  credit_left -= cost;
274  }
275 
276  // try to send that much
277 
278  std::size_t wrote = 0;
279 
280  if (can_write > 0) {
281  LOG(VERBOSE) << "Try to write " << can_write << " bytes to server";
282 
283  auto data = write_buffer_.Front(can_write);
284  for (const base::BufferView& slice : { data.first, data.second }) {
285  if (slice.size == 0)
286  break;
287 
288  std::size_t ret;
289  try {
290  ret = socket_->Write(slice.data, slice.size);
291  } catch (const event::Socket::Exception& e) {
292  ConnectionLost(e.what());
293  return;
294  }
295 
296  wrote += ret;
297  if ((base::BufferSize)ret != slice.size)
298  break; // couldn't fit all of it in
299  }
300 
301  if (wrote == 0)
302  return; // wait for next round, write still started
303  }
304 
305  // pop off what we managed to write
306 
307  write_buffer_.Pop(wrote);
308 
309  std::size_t pop = wrote;
310  while (pop > 0) {
311  CHECK(!write_queue_.empty());
312  auto& msg = write_queue_.front();
313  std::size_t first_bytes = msg.first;
314  if (first_bytes <= pop) {
315  pop -= first_bytes;
316  write_credit_ -= 10 * msg.first + msg.second;
317  write_queue_.pop_front();
318  } else {
319  msg.first -= pop;
320  write_credit_ -= 10 * pop;
321  break;
322  }
323  }
324 
325  // if we couldn't fit everything we can afford, start a new write
326 
327  if (wrote < can_write) {
328  socket_->StartWrite();
329  return;
330  }
331 
332  // if we had anything left, see how long we need to wait to afford that
333 
334  if (!write_queue_.empty()) {
335  const auto& msg = write_queue_.front();
336  int cost = 10 * msg.first + msg.second;
337  int debt = std::max(cost - write_credit_, 0);
338  write_credit_timer_ = loop_->Delay(std::chrono::milliseconds(debt), &write_credit_timer_callback_);
339  }
340 }
