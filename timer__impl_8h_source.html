49 void DefaultTimerFd::Arm(TimerDuration delay) {
50  struct itimerspec next_timer = {{0, 0}, {0, 0}};
51  next_timer.it_value.tv_nsec = std::chrono::duration_cast<std::chrono::nanoseconds>(delay).count();
52  next_timer.it_value.tv_sec = next_timer.it_value.tv_nsec / 1000000000;
53  next_timer.it_value.tv_nsec %= 1000000000;
54 
55  if (timerfd_settime(fd_, 0, &next_timer, nullptr) == -1)
56  throw Exception("timerfd_settime", errno);
57 }

115 std::pair<typename Timer<PeriodicT, OneshotT>::Request*, PeriodicT*> Timer<PeriodicT, OneshotT>::AddPeriodic(TimerDuration rate, Args&&... args) {
116  if (auto req = periodic_.find(rate); req != periodic_.end())
117  return std::pair(req->second, &req->second->data);
118 
119  PeriodicRequest* new_req = requests_.insert(std::make_unique<PeriodicRequest>(NextPeriod(rate), rate, std::forward<Args>(args)...));
120  periodic_.emplace(rate, new_req);
121  queue_.insert(new_req);
122  UpdateTimer();
123 
124  return std::pair(new_req, &new_req->data);
125 }

152 template <typename PeriodicT, typename OneshotT>
153 template <typename F>
154 void Timer<PeriodicT, OneshotT>::Poll(F f) {
155  timerfd_->Wait();
156 
157  TimerPoint now = timerfd_->now();
158  while (!queue_.empty()) {
159  Request* req = *queue_.begin();
160  if (req->target > now)
161  break;
162 
163  queue_.erase(queue_.begin());
164 
165  if (req->periodic)
166  f(&static_cast<PeriodicRequest*>(req)->data, nullptr);
167  else
168  f(nullptr, &static_cast<OneshotRequest*>(req)->data);
169 
170  if (req->periodic) {
171  req->target = NextPeriod(static_cast<PeriodicRequest*>(req)->rate);
172  queue_.insert(req);
173  } else {
174  requests_.erase(req);
175  }
176  }
177 
178  UpdateTimer();
179 }

188 template <typename PeriodicT, typename OneshotT>
189 TimerPoint Timer<PeriodicT, OneshotT>::NextPeriod(TimerDuration rate) {
190  auto realtime_now = std::chrono::system_clock::now();
191  auto count_now = realtime_now.time_since_epoch().count();
192  auto count_rate = std::chrono::duration_cast<std::chrono::system_clock::duration>(rate).count();
193  auto count_delay = count_rate - ((count_now % count_rate) + count_rate) % count_rate;
194  return timerfd_->now() + std::chrono::system_clock::duration(count_delay);
195 }

199  using namespace std::chrono_literals;
200  constexpr TimerDuration kSlack = 1ms;
201 
202  if (queue_.empty()) // TODO: disarm timer if armed? maybe keep track.
203  return;
204

206  if (delay < kSlack)
