29 /**
30  * Base class for callback interfaces.
31  *
32  * Any callback interfaces registered to the callback containers (CallbackPtr, CallbackSet) must
33  * have this class as a base class. They should probably use virtual inheritance, in case the client
34  * wants to use a single class to implement more than one callback interface.
35  *
36  * This class contains the support code for the callback to unregister itself from any callback
37  * container it's been added to when it gets destroyed. Conversely, callback containers know how to
38  * undo the linkage from their callbacks when they themselves get destroyed.
39  */
40 class Callback {
41  public:
42  virtual ~Callback() {
43  DetachCallback();
44  }

46  /** Immediately detaches the callback from any containers it has been added to. */
47  void DetachCallback() {
48  for (auto&& item : containers_)
49  item.first->Unregister(this, item.second);
50  containers_.clear();
51  }

53  private:
54  std::map<internal::CallbackContainer*, void*> containers_;
55 
56  void RegisterContainer(internal::CallbackContainer* container, void* data) {
57  auto [iter, inserted] = containers_.try_emplace(container, data);
58  (void) iter;
59  CHECK(inserted);
60  }

63  bool erased = containers_.erase(container);
64  CHECK(erased);

100  */
101  void Set(Iface* callback, bool owned = false) {

103  Clear();

109  /**

111  *
112  * If the callback was owned, it's destroyed. Otherwise it's simply unlinked from this
113  * container. If the container was empty, nothing happens.
114  */
115  void Clear() {
116  if (callback_) {
117  if (owned_)
118  delete callback_;
119  else

396 /**
397  * Helper template to create wrappers from `std::function` to callbacks.
398  *
399  * Typically used with the `CALLBACK_*` series of macros.
400  *
401  * Example for a single-method interface:
402  *
403  * struct Foo {
404  * virtual void Bar(int) = 0;
405  * };
406  * CALLBACK_F1(FooF, Foo, Bar, int);
407  *
408  * This defines a `FooF` class, which derives from `Foo`. The class has a single-argument
409  * constructor accepting a `const std::function<void(int)>&`, which it uses to initialize a
410  * `std::function` object. When the `Bar` method is called, it is delegated to the stored function.
411  *
412  * Example for a two-method interface:
413  *
414  * struct Foo {
415  * virtual void Bar(int, float) = 0;
416  * virtual void Baz(char) = 0;
417  * };
418  * CALLBACK_F(FooF, Foo) {
419  * CALLBACK_C(FooF);
420  * CALLBACK_M2(0, Bar, int, float);
421  * CALLBACK_M1(1, Baz, char);
422  * };
423  *
424  * This defines a `FooF` class, which derives from `Foo`. The class has a two-argument constructor,
425  * accepting a `const std::function<void(int,float)>&` and a `const std::function<void(char)>&`,
426  * which it uses to initialize two `std::function` objects. When the `Bar` method is called, it's
427  * delegated to the first function. When the `Baz` method is called, it's delegated to the second
428  * one.
429  */
430 template<typename... Funcs>
431 class CallbackF {
432  protected:
433  std::tuple<std::function<Funcs>...> f_;
434  public:
435  template<typename... CFuncs>
436  CallbackF(CFuncs&&... args) : f_(std::forward<CFuncs>(args)...) {}
437 };

439 /** Starts a callback wrapper class definition, see CallbackF. */
440 #define CALLBACK_F(Name, Iface, ...) struct Name : public Iface, public ::base::CallbackF<__VA_ARGS__>
441 /** Defines a callback wrapper constructor, see CallbackF. */
442 #define CALLBACK_C(Name) template<typename... CFuncs> explicit Name(CFuncs&&... args) : CallbackF(std::forward<CFuncs>(args)...) {}
443 /** Defines a single-argument method in a callback wrapper class, see CallbackF. */
444 #define CALLBACK_M1(n, Method, T) void Method(T a) override { std::get<n>(f_)(std::forward<T>(a)); }
445 /** Defines a single-argument method in a callback wrapper class, see CallbackF. */
446 #define CALLBACK_M2(n, Method, T1, T2) void Method(T1 a, T2 b) override { std::get<n>(f_)(std::forward<T1>(a), std::forward<T2>(b)); }
