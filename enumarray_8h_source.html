13 /**
14  * An array for mapping (dense) enumeration constants to a type.
15  *
16  * This type is expected to be used in constexpr declarations, in which case it will turn into the
17  * raw bytes of an array of \p T of length \p N.
18  *
19  * \p N should be one more than the highest-valued constant of \p Enum. Including higher values in
20  * the array initializer will be detected at compile time, but looking them up will not.
21  */
22 template <typename Enum, typename T, int N>
23 struct EnumArray {
24  /** Array storing the items. */
25  T items[N];
26  /**
27  * Constructs a new array from a list of { constant, value } pairs.
28  *
29  * If the \p def argument is provided, uninitialized elements will be set to that value.
30  */
31  constexpr EnumArray(std::initializer_list<std::pair<Enum, T>> data, T def=T()) : items{def} {
32  for (int i = 1; i < N; i++)
33  items[i] = def;
34  for (auto&& item : data) {
35  if (static_cast<int>(item.first) >= N)
36  throw "EnumArray index larger than declared size";
37  items[static_cast<int>(item.first)] = item.second;
38  }
39  }
40  /** Looks up the value corresponding to an enum constant in the array. */
41  constexpr T operator[](Enum key) const {
42  return items[static_cast<int>(key)];
43  }
44  /** Returns the (compile-time constant) size of the array. */
45  constexpr int size() const { return N; }
46 };
