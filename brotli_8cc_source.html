33 bool BrotliInputStream::Next(const void** data, int* size) {
34  /// \todo remove buffer resizing, trust that brotli lib can cope
35 
36  // decompress data, unless there's some pending from a BackUp call
37 
38  while (data_ == data_end_) {
39  BrotliDecoderResult brotli_result;
40 
41  // try to decompress pending input (if any)
42 
43  while (true) {
44  data_ = &data_buffer_[0];
45  data_end_ = data_;
46  data_space_ = data_buffer_.size();
47 
48  brotli_result = BrotliDecoderDecompressStream(
49  brotli_.get(),
50  &stream_chunk_available_, &stream_chunk_,
51  &data_space_, &data_end_,
52  nullptr);
53 
54  if (data_end_ > data_ || brotli_result != BROTLI_DECODER_RESULT_NEEDS_MORE_OUTPUT)
55  break; // got some, or allocating more output size won't help
56  if (data_buffer_.size() >= kMaxBufferSize)
57  return false; // impossible: ridiculous output buffer needed
58  data_buffer_.resize(2 * data_buffer_.size());
59  }
60 
61  if (data_end_ > data_) {
62  break; // got some data, deliver to client
63  } else if (brotli_result == BROTLI_DECODER_RESULT_SUCCESS || brotli_result == BROTLI_DECODER_RESULT_ERROR) {
64  // unread input that wasn't consumed
65  if (stream_chunk_available_ > 0) {
66  stream_->BackUp(stream_chunk_available_);
67  stream_chunk_ = nullptr;
68  stream_chunk_available_ = 0;
69  }
70  return false; // decoder says no more output is forthcoming
71  } else if (brotli_result == BROTLI_DECODER_RESULT_NEEDS_MORE_INPUT) {
72  // pull more input from underlying stream
73  while (stream_chunk_available_ == 0) {
74  const void *chunk;
75  int size;
76  if (!stream_->Next(&chunk, &size))
77  return false; // no more to give
78  if (size == 0)
79  continue; // try again to get some actual data
80  stream_chunk_ = static_cast<const std::uint8_t*>(chunk);
81  stream_chunk_available_ = size;
82  }
83  continue; // try to decompress again, with new data
84  } else {
85  return false; // impossible: weird return value
86  }
87  }
88 
89  byte_count_ += data_end_ - data_;
90 
91  *data = data_;
92  *size = data_end_ - data_;
93  data_ = data_end_;
94  return true;
95 }
